      axis lines = left,

    ]
    % The red parabola is defined
    \addplot[
      domain = -10:10,
      samples = 100,
      color = red,
    ] {x^2 - 2*x - 1};
    \addlegendentry{\(x^2 - 2x - 1\)}
    % The blue parabola is defined

      samples = 100,
      color = blue,
    ] {x^2 + 2*x + 1};
    \addlegendentry{\(x^2 + 2x + 1\)}
    \end{axis}
